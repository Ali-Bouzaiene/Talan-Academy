fonction triInsertion(ref T :tableau[1..N] chaine de cartères,):vide;
var i,j,k:entier;
T=["Sirius", "Rigel", "Bételgeuse", "Arcturus", "Aldébaran", "Véga", "Deneb", "Altaïr"];
Début
        pour i de 1 à taille(T) - 1
            k ← T[i];                   
            j ← i;                               
            tant que j > 0 et T[j- 1] > k
                     T[j] ← T[j-1];
                     j ← j - 1;
                     T[j] ← k ;
            fin tanque
        fin pour
Fin
Fin fonction
